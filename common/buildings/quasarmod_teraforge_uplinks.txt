    base_buildtime = @b3_time
    planet_limit = 1
    category = manufacturing

    empire_limit = {
        base = 0
        modifier = {
            exists = owner
            weight = owner.modifier:building_quasarmod_teraforge_uplink_country_max
        }
    }

    building_sets = {
        foundry
        industrial
    }

    prerequisites = {
        tech_quasarmod_hyperquasaric_teraforge
    }

    show_tech_unlock_if = { always = no }

    potential = {
        exists = owner
        owner = { is_wilderness_empire = no }
        NOR = {
            has_modifier = resort_colony
            has_modifier = slave_colony
        }

    destroy_trigger = {
        exists = owner

            has_modifier = resort_colony
            has_modifier = slave_colony

        }
    }

    allow = {
        has_major_upgraded_capital = yes

        category = planet_buildings

            energy = 50
        }
    }

    triggered_desc = {
        trigger = {
            exists = owner
            owner = {
                is_gestalt = no
                is_catalytic_empire = no
            }
        }
        text = building_teraforge_uplink_alloys_triggered_desc
    }
    triggered_desc = {
        trigger = {
            exists = owner
            owner = {
                is_gestalt = no

            }

    triggered_desc = {
